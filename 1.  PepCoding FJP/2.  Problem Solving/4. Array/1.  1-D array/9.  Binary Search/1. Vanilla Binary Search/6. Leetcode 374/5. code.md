public class Solution extends GuessGame {
    public int guessNumber(int n) {
        int l = 0;
        int r = n;
        int num;
        
        while(l <= r)
        {
            num = l + ((r - l)/2);
            if(guess(num) == -1){
                r = num -1;
            }
            else if(guess(num) == 1){
                l = num + 1;
            }
            else{
                return num;
            }
        }

        return -1;
    }
}
