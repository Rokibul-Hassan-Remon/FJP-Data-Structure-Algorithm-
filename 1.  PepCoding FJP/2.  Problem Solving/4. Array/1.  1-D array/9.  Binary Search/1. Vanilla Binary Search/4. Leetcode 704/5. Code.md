```java title:"Leetcode:704 Binary Search"
class Solution 
{
    public int search(int[] nums, int target)
    {
      int left =0;
      int right = nums.length-1;
      int mid = -1;

      while(left <= right){
        mid = left + ((right - left)/2);
        
        if(target == nums[mid]){
            return mid;
        }
        
        if( target > nums[mid]){
            left = mid + 1;
        }
        else if(target < nums[mid]){
            right = mid - 1;
        }        
      }  
      
      return -1;
    }
}
```
