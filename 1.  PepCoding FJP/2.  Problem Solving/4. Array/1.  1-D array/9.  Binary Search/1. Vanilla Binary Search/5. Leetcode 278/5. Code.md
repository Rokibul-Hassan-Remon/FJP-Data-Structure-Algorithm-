public class Solution extends VersionControl {
	public static int badVersions;
	  
	// It's my own function as a dummy function
	public static boolean isBadVersion(int x){  
	    if(badVersions <= x){  
	        return true;  
	    }  
	    return false;   
	}

	// It's a Leetcode original code which was given
    public int firstBadVersion(int n) 
    {
        int l = 1;
        int r = n;
        int result = n;

        while (l <= r)
        {
            int mid = l + ((r - l) / 2);

            if (isBadVersion(mid) == true)
            {
                if (result > mid) {
                    result = mid;
                }
                r = mid - 1;
            } 
            else 
            {
                l = mid + 1;
            }
        }
        
        return result;
    }

	public static void main(String[] args){  
	    Scanner scn = new Scanner(System.in);  
	    Println("Input the current version number");  
	    int n = scn.nextInt();  
	  
	    Println("Input the last how many version you got the bad version");  
	    badVersions = scn.nextInt();  
	  
	    int result = firstBadVersion(n);  
	    Println("First bad version : "+result);  
	}

	public static void Println(String str)  
    {  
        System.out.println(str);  
    }  
    
    public static void Print(String str)

    }  
}
