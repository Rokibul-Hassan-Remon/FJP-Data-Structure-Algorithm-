public class GraphBiPariti {  
    public int vertex;  
    public int level;  
  
    public GraphBiPariti(int v, int l){  
        vertex = v;  
        level = l;  
    }  
}

	public class Main {  
		public static boolean ByPariti(ArrayList<Edge>[] graph,  int src, int[] isVisited){  
	    ArrayDeque<GraphBiPariti> bfs = new ArrayDeque<>();  
	  
	    bfs.add(new GraphBiPariti(src, 0));  
	  
	    while( !bfs.isEmpty() ) {  
	        // remove  
	        var top = bfs.remove();  
	  
	        if(isVisited[top.vertex] != -1){  
	            if( isVisited[top.vertex] !=  top.level){  
	                return false;  
	            }  
	        }  
	  
	        isVisited[top.vertex] = top.level;  
	  
	        for (Edge each : graph[top.vertex]) {  
	            if(isVisited[each.neighbour] == -1)  
	                bfs.add(new GraphBiPariti(each.neighbour, top.level+1));  
	        }  
	  
	    }  
	  
	    return true;  
	}  
  
	public static boolean IsGraphBiParity(ArrayList<Edge>[] graph){  
	    int[] isVisited = new int[graph.length];  
	    Arrays.fill(isVisited, -1);  
	  
	    for(int i = 0; i < isVisited.length; i++){  
	        if( isVisited[i] == -1 &&  !ByPariti(graph, i, isVisited) ){  
	            return false;  
	        }  
	    }  
	  
	    return  true;  
	}
