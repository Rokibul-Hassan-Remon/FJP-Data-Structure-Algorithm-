public class Edge {  
    int source;  
    int neighbour;  
    int weight;  
  
    public Edge(int source, int neighbour, int weight){  
        this.source = source;  
        this.neighbour = neighbour;  
        this.weight = weight;  
    }  
}

public class Main {  
	public static class PathAndWeight{  
	    String path;  
	    int weight;  
	  
	    public PathAndWeight(int weight, String path){  
	        this.weight = weight;  
	        this.path = path;  
	    }  
	}  
	public static ArrayList<PathAndWeight> list = new ArrayList<>();  
	  
	public static void FindKthLeargestPath(ArrayList<Edge>[] graph, int src, int dest, boolean[] isVisited, String path, int totalWeight){  
	    isVisited[src] = true;  
	    if(src == dest){  
	        // Ceil path weight condition  
	        list.add( new PathAndWeight(totalWeight, path));  
	        isVisited[src] = false;  
	  
	        return;  
	    }  
	  
	    for(var each : graph[src]){  
	        if(!isVisited[each.neighbour])  
	            FindKthLeargestPath(graph, each.neighbour, dest, isVisited, path+" "+each.neighbour, totalWeight + each.weight);  
	    }  
	  
	    isVisited[src] = false;  
	}  
	  
	public static void main(String[] args) throws Exception {  
	    BufferedReader bfr = new BufferedReader(new InputStreamReader(System.in));  
	    int vertices = Integer.parseInt(bfr.readLine().trim());  
	  
	    ArrayList<Edge>[] adjacencyList = new ArrayList[vertices];  
	    for(int i = 0; i < adjacencyList.length; i++){  
	        adjacencyList[i] = new ArrayList<Edge>();  
	    }  
	  
	    int edges = Integer.parseInt(bfr.readLine());  
	    for(int i = 0; i < edges; i++){  
	        String[] str = bfr.readLine().split(" ");  
	        int v1 = Integer.parseInt(str[0]);  
	        int v2 = Integer.parseInt(str[1]);  
	        int wght = Integer.parseInt(str[2]);  
	  
	        adjacencyList[v1].add( new Edge(v1, v2, wght));  
	        adjacencyList[v2].add( new Edge(v2, v1, wght));  
	    }  
	  
	    int stratingPoint = Integer.parseInt(bfr.readLine().trim());  
	    int endingPoint = Integer.parseInt(bfr.readLine().trim());  
	    int kth = Integer.parseInt(bfr.readLine().trim());  
	    boolean[] isVisited = new boolean[vertices];  
	  
	    FindKthLeargestPath(adjacencyList, stratingPoint, endingPoint, isVisited, stratingPoint+"", 0);  
	    Collections.sort(list, new Comparator<PathAndWeight>(){  
	        @Override  
	        public int compare(PathAndWeight a, PathAndWeight b){  
	            return b.weight - a.weight;  
	        }  
	    });  
	    var target = list.get(kth);  
	    Println("Path: "+target.path+" Weight: " + target.weight);  
	}
  
	public static void Println(String str) {  
		System.out.println(str);  
	}  
	public static void Print(String str) {  
		System.out.print(str);  
	}  
}
