```java title:"Find the connected Vertices group wise"
public class Edge {  
    int source;  
    int neighbour;  
    int weight;  
  
    public Edge(int source, int neighbour, int weight){  
        this.source = source;  
        this.neighbour = neighbour;  
        this.weight = weight;  
    }  
}

public class Main {  
	public static ArrayList<ArrayList<Integer>> componentList = new ArrayList<>();  
	public static ArrayList<Integer> ConnectedComponents(ArrayList<Edge>[] graph, boolean[] isVisited, int src) {  
	    isVisited[src] = true;  
	    ArrayList<Integer> path = new ArrayList<Integer>();  
	  
	    for(var each: graph[src]){  
	        if( !isVisited[each.neighbour] ){  
	            ArrayList<Integer> visitedPath = ConnectedComponents(graph, isVisited, each.neighbour);  
	            path.add(src);  
	            path.addAll(visitedPath);  
	        }  
	    }  
	  
	    if(path.isEmpty() == true)  
	         path.add(src);  
	  
	    return path;  
	}  
	  
	public static void main(String[] args) throws Exception {  
	    BufferedReader bfr = new BufferedReader(new InputStreamReader(System.in));  
	    int vertices = Integer.parseInt(bfr.readLine().trim());  
	  
	    ArrayList<Edge>[] adjacencyList = new ArrayList[vertices];  
	    for(int i = 0; i < adjacencyList.length; i++){  
	        adjacencyList[i] = new ArrayList<Edge>();  
	    }  
	  
	    int edges = Integer.parseInt(bfr.readLine());  
	    for(int i = 0; i < edges; i++){  
	        String[] str = bfr.readLine().split(" ");  
	        int v1 = Integer.parseInt(str[0]);  
	        int v2 = Integer.parseInt(str[1]);  
	        int wght = Integer.parseInt(str[2]);  
	  
	        adjacencyList[v1].add( new Edge(v1, v2, wght));  
	        adjacencyList[v2].add( new Edge(v2, v1, wght));  
	    }  
	  
	    boolean[] isVisited = new boolean[vertices];  
	    componentList = new ArrayList<>();  
	    for(int i = 0; i < isVisited.length; i++){  
	        if( !isVisited[i] ){  
	            ArrayList<Integer> connectedPath = ConnectedComponents(adjacencyList, isVisited, i);  
	            componentList.add(connectedPath);  
	        }  
	    }  
	  
	    Println(componentList+" ");  
	}
  
	public static void Println(String str) {  
		System.out.println(str);  
	}  
	public static void Print(String str) {  
		System.out.print(str);  
	}  
}
```
