	public static void IterativeWayPreAndPostOrderTreeTraversal(TreeNode root){  
	    String preOrder="";  
	    String postOrder="";  
	  
	    Stack<Pair> temp = new Stack<>();  
	  
	    Pair p = new Pair(root, -1);  
	    temp.push(p);  
	  
	    while(temp.size() > 0){  
	        p = temp.peek();  
	  
	        if(p.status == -1)  
	        {  
	            p.status++;  
	            preOrder += p.node.data+" ";  
	        }  
	        else if(p.status < p.node.children.size()){  
	            TreeNode child = p.node.children.get(p.status);  
	            Pair curPair = new Pair(child,-1);  
	            temp.push(curPair);  
	            p.status++;  
	        }  
	        else if(p.status == p.node.children.size()){  
	            p = temp.pop();  
	            postOrder += p.node.data+" ";  
	        }  
	    }  
	  
	    Println(preOrder+" ");  
	    Println(postOrder+" ");  
	}  
	  
	public static void main(String[] args) {  
	    TreeNode root = GetTree();  
	    IterativeWayPreAndPostOrderTreeTraversal(root);  	  
	}

	    System.out.println(str);  
	}
