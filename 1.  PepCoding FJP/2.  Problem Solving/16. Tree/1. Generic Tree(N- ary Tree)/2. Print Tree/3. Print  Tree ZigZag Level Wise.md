	public static  void ZigZagLevelWiese(TreeNode root){   
	    Stack<TreeNode> parents = new Stack<>();  
		Stack<TreeNode> childs = new Stack<>();  
		int level = 0;  
		parents.add(root);  
		  
		while(parents.size() > 0){  
		    TreeNode element = parents.pop();  
		    Print(element.data+" ");  
		    
		    if(level % 2 == 0)  
		        for(var child: element.children){  
		            childs.push(child);  
		        }  
		    else{  
		        for(int i = element.children.size() - 1; i >= 0; i--){  
		            childs.push(element.children.get(i));  
		        }  
		    }  
		  
		    if(parents.size() == 0){  
		        parents = childs;  
		        childs = new Stack<>();  
		        Println("");  
		        level++;  
		    }  
		}
	}
