	public static List<Integer> NodeToRootPath(TreeNode root, int searchValue)  
	{  
	    List< Integer > list = new ArrayList<>();  
	    ArrayDeque< Pair > stack = new ArrayDeque<>();  
	    
	    if(root == null){  
	        return list;  
	    }  
	    
	    Pair p = new Pair(root,0);  
	    stack.push(p); 
	     
	    while( !stack.isEmpty() ){  
	        Pair temp = stack.peek();  
	        
	        if(temp.status == 1)  
	        {  
	            stack.pop();  
	            continue;  
	        }  
	        
	        temp.status++;  
	        
	        if(temp.node.data == searchValue)  
	            break;  
	            
	        if(temp.node.children != null){  
	            ArrayList<TreeNode> chicks = temp.node.children;  
	            
	            for(int i = chicks.size() - 1; i >= 0; i--){  
	             stack.push( new Pair(chicks.get(i),0 ) );  
	            }  
	         }  
	    }
	      
	    while( !stack.isEmpty() ){  
	        Pair addOrNot = stack.pop();  
	          
	        if(addOrNot.status == 1)  
	            list.add(addOrNot.node.data);  
	    }  
	    
	    return list;  
	}
