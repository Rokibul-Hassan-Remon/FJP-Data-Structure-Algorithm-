```java title:"Mirror Image of a tree"
public void TreeImage(TreeNode node){
	//node pre
	for(var each : node.children){
		TreeImage(each);
	}

	// node post
	ArrayList<TreeNode> list = node.children;
	for(int i = list.Size() - 1, j = 0; i > j; i--,j++){
		TreeNode temp = list[j];
		list[j] = list[i];
		list[i] = temp;
	}
}
```
