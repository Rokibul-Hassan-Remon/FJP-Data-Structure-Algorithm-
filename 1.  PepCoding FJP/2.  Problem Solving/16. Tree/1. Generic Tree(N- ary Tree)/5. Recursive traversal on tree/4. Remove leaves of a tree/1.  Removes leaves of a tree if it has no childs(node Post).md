	public static void RemovesLeaves(TreeNode node){ 
		//node Pre 
	    for(int i = node.children.size() - 1; i >= 0; i--){  
	        TreeNode temp = node.children.get(i);  
	        
	        if(temp.children.size() == 0){  
	            node.children.remove(i);  
	        }  
	    }  
	    
		//node Post
	    for(var each : node.children){  
	        RemovesLeaves(each);  
	    }
