```java title:"Basic structure of recursive on tree."

public void BasicRecursiveStyle(TreeNode node){
	// Node pree
	Println("Node-pre "+node.data);

	// loops,before calling is pre, after calling is post
	for(var each : node.children){
		// edge pre
		Println("Edge-pre "+node.data+" - "+each.data);
		
		BasicRecursiveStyle(each);
		
		// edge post
		Println("Edge-post "+node.data+" - "+each.data);
	}

	Println("Node-post "+node.data);
	// node post
}

	public static void Println(String str) {  
	    System.out.println(str);  
	}
	  
	public static void Print(String str) {  
	    System.out.print(str);  
	}
```
