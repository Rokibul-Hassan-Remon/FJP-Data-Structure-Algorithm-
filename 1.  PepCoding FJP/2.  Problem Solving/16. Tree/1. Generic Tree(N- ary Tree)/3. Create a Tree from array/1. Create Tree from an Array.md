public class TreeNode {  
    public int data;  
    public ArrayList<TreeNode> children = new ArrayList<TreeNode>();  
  
    public TreeNode(int value){  
        data = value;  
    }  
  
    @Override  
    public String toString() {  
        return String.valueOf(this.data);  
    }  
}

import java.util.*;  
  
public class Main {
	public static TreeNode Construt(int[] array){  
	    Stack<TreeNode> store = new Stack<>();  
	    TreeNode root ;
	    for(int i = 0; i < array.length; i++){  
	        if(array[i] != -1)  
	        {  
	            TreeNode newNode = new TreeNode(array[i]);  
	            store.push((newNode));  
	        }  
	        else{  
	            TreeNode temp = store.pop();  
	            if(store.size() > 0) {  
	                var node = store.peek();  
	                node.children.add(temp);  
	            }else{
		            root = temp; 
	            }  
	        }  
	    }

		return root;
    }
