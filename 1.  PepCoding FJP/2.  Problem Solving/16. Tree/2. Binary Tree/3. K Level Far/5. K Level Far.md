public class BinaryTree {  
    public int data;  
    public BinaryTree left;  
    public BinaryTree right;  
  
    public BinaryTree(int data, BinaryTree left, BinaryTree right){  
        this.data = data;  
        this.left = left;  
        this.right = right;  
    }  
}

import java.util.*;  
  
public class Main {
  
    public static BinaryTree GetBinaryTree()

    public static void KthDown(BinaryTree node, int k, BinaryTree block)
    {
      
        if(node == null || k < 0 || block == node)  
            return;  
  
        if(k == 0){  
            Println(node.data+" ");  
            return;  
        }  
  
        if (node.left != null)  
            KthDown(node.left, k - 1, block);  
  
        if (node.right != null)  
            KthDown(node.right, k - 1, block);  
    }  
    
	public static List<BinaryTree> NodeToRooTPath(BinaryTree node, int search) 
	{
	  
        ArrayList<BinaryTree> list = new ArrayList<>();  
  
        if(node == null){  
            return list;  
        }  
  
        if(search == node.data){  
            list.add(node);  
  
            return list;  
        }  
  
        List<BinaryTree> leftPath = NodeToRooTPath(node.left, search);  
        list.addAll(leftPath);  
  
        List<BinaryTree> rightPath = NodeToRooTPath(node.right, search);  
        list.addAll(rightPath);  
  
        if(!list.isEmpty())  
            list.add(node);  
  
        return list;  
    }  
    
    public static void NodeToKLevelFar(List<BinaryTree> list, int dstncAway)
    {
      
        for(int i = 0; i < list.size(); i++){  
            BinaryTree node = list.get(i);  
  
            BinaryTree block = ( i == 0)? null : list.get(i -1);  
  
            if( dstncAway - i >= 0 )  
                KthDown(node, dstncAway - i, block);  
        }  
    }  
      
    public static void main(String[] args)
    { 
     
       BinaryTree root = GetBinaryTree();  
       int dstncAway = 3;  
  
       List<BinaryTree> list = NodeToRooTPath(root, 70);  
  
       Print("Node to root path: ");  
       for(var each : list){  
           Print(each.data+" ");  
       }  
       Println("");  
  
       NodeToKLevelFar(list,dstncAway);  
    }
