    public int data;

    public BinaryTree right;  
     
    public BinaryTree(int data, BinaryTree left, BinaryTree right){  
        this.data = data;  
        this.left = left;  
        this.right = right;  
    }  
}

    private static int preorderIndex = 0; // instance-level state  
	private static Map<Integer, Integer> inorderIndexMap;

	    inorderIndexMap = new HashMap<>();

	        inorderIndexMap.put(inorder[i], i);

	private static BinaryTree constructTree(int[] preorder, int start, int end) {  
	    if (start > end)  
	            return null;

	    root.left = constructTree(preorder, start, inorderRootIndex - 1);  
	    root.right = constructTree(preorder, inorderRootIndex + 1, end);  
	  
	    return root;  
	}
