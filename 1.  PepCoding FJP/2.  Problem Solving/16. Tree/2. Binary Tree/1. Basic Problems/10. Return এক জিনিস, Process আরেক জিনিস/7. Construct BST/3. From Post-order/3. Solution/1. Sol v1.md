```java title:"Construct BST from Post-order Traversal list"

	private static int postOrderIndex = 0; /// when initialize it will be postOrder.length - 1  
	public static BinaryTree BSTFromPostOrder(int[] postOrder, int min, int max){  
	    if(postOrderIndex >= postOrder.length || max < postOrder[postOrderIndex] || min > postOrder[postOrderIndex]) return null;  
	  
	    BinaryTree node = new BinaryTree(postOrder[postOrderIndex]);  
	    postOrderIndex--;  
	  
	    node.right = BSTFromPostOrder(postOrder, postOrder[postOrderIndex], max);  
	    node.left = BSTFromPostOrder(postOrder, min, postOrder[postOrderIndex]);  
	  
	    return node;  
	}
```
