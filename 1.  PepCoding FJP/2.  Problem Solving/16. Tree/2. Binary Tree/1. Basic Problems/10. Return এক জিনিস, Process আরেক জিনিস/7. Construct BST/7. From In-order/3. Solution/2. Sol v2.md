	public static BinaryTree buildBSTFromInorder(int[] inorder, int start, int end) {
	    // Base case: invalid range
	    if (start > end) return null;
	
	    // Calculate middle safely to avoid integer overflow
	    int mid = start + (end - start) / 2;
	
	    // Create current root node
	    BinaryTree root = new BinaryTree(inorder[mid]);
	
	    // Build left subtree
	    root.left = buildBSTFromInorder(inorder, start, mid - 1);
	
	    // Build right subtree
	    root.right = buildBSTFromInorder(inorder, mid + 1, end);
	
	    return root;
	}
