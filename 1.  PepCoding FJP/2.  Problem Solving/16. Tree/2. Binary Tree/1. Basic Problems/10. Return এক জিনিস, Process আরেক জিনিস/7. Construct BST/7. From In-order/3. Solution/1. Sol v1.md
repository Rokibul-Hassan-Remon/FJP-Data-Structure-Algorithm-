```java title:"Construct BST to In-order list"

	public static BinaryTree TreeConstructFromInOrder(int[] inorder, int start, int end){  
	    if(start > end || start < 0 || end >= inorder.length){  
	       return null;  
	    }  
	  
	    int middle = 0;  
	    middle = start + ( end - start)/2;  
	  
	    BinaryTree node = new BinaryTree( inorder[middle] );  
	    node.left = TreeConstructFromInOrder(inorder, start, middle - 1);  
	    node.right = TreeConstructFromInOrder(inorder, middle + 1, end);  
	  
	    return node;  
	}
```
