```java title:"Construct BST from Pre-order Traversal List"

	private static int index = 0;  
	public static BinaryTree BSTFromPreOrder(int[] preOrder, int min, int max){  
	    if(index >= preOrder.length || max < preOrder[index] || min > preOrder[index]) return null;  
	  
	    BinaryTree node = new BinaryTree(preOrder[index]);  
	    index++;  
	    node.left = BSTFromPreOrder(preOrder, min, preOrder[index]);  
	    node.right = BSTFromPreOrder(preOrder, preOrder[index], max);  
	  
	    return node;  
	}
```
