public class BinaryTree {  
    public int data;  
    
    public BinaryTree left;  
    public BinaryTree right;  
  
    public BinaryTree(int data, BinaryTree left, BinaryTree right){  
        this.data = data;  
        this.left = left;  
        this.right = right;  
    }  
}

import java.util.*;  
  
public class Main {
    public static int maxDiameter = 0;
    
    public static BinaryTree GetBinaryTree()
    {     
        BinaryTree root = new BinaryTree(10, null, null);  
  
        root.left = new BinaryTree(20,null,null);  
        root.right = new BinaryTree(30, null, null);  
  
        BinaryTree  lftLvl = root.left;  
        BinaryTree  rghtLvl = root.right;  
  
        lftLvl.left = new BinaryTree(40, null, null);  
        lftLvl.right = new BinaryTree(50, null, null);  
  
        rghtLvl.left = new BinaryTree(60,null,null);  
        rghtLvl.right = new BinaryTree(70,null,null);  
  
        lftLvl =  lftLvl.left;  
        lftLvl.left = new BinaryTree(80, null, null);  
        lftLvl.right = new BinaryTree(90, null, null);  
  
        lftLvl = lftLvl.right;  
        lftLvl.left = new BinaryTree(100, null, null);  
        lftLvl.right = new BinaryTree(110, null, null);  
  
        rghtLvl = rghtLvl.left;  
        rghtLvl.left = new BinaryTree(120, null, null);  
        rghtLvl.right = new BinaryTree(130, null, null);  
  
        rghtLvl = rghtLvl.right;  
        rghtLvl.left = new BinaryTree(140, null, null);  
        rghtLvl.right = new BinaryTree(150, null, null);  
  
        return  root;  
    }  
    
	public static int maxDepth(BinaryTree node) {  
        if(node == null)  
            return 0;  
            
        int leftH = maxDepth(node.left);  
        int rightH = maxDepth(node.right);  
        
        int throughNode = leftH + rightH;  
        maxDiameter = Math.max(maxDiameter, throughNode);  
        
        return ( 1 + Math.max(leftH, rightH) );  
    }  
  
    public static int DiameterOfBinaryTree(BinaryTree root) {  
        maxDiameter = 0;  
        maxDepth(root);  
        
        return maxDiameter;  
    } 
    
    public static void main(String[] args)
    { 
       BinaryTree root = GetBinaryTree();  
       
       int diameter = DiameterOfBinaryTree(root);  
	   System.out.println("Diameter of tree: " + diameter);
    }  
    
    public static void Println(String str)

        System.out.println(str);  
    } 
    
    public static void Print(String str)
    {
        System.out.print(str);  
    }  
}
