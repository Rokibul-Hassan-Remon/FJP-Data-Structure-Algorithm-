boolean isBST(Node root) {
    Node curr = root, prev = null;
    int lastValue = Integer.MIN_VALUE;
    boolean firstNode = true;

    while (curr != null) {
        if (curr.left == null) {
            if (!firstNode && curr.data <= lastValue)
                return false;
            firstNode = false;
            lastValue = curr.data;
            curr = curr.right;
        } else {
            Node pre = curr.left;
            while (pre.right != null && pre.right != curr)
                pre = pre.right;

            if (pre.right == null) {
                pre.right = curr;
                curr = curr.left;
            } else {
                pre.right = null;
                if (curr.data <= lastValue)
                    return false;
                lastValue = curr.data;
                curr = curr.right;
            }
        }
    }
    return true;
}
