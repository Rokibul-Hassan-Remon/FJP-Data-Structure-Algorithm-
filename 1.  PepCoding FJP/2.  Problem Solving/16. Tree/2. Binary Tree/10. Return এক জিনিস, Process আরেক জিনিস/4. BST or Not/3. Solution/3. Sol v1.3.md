	boolean isBST(Node root) {
	    Stack<Node> stack = new Stack<>();
	    Node curr = root;
	    Node prev = null;
	
	    while (curr != null || !stack.isEmpty()) {
	        while (curr != null) {
	            stack.push(curr);
	            curr = curr.left;
	        }
	
	        curr = stack.pop();
	        if (prev != null && curr.data <= prev.data)
	            return false;
	        prev = curr;
	        curr = curr.right;
	    }
	    return true;
	}
