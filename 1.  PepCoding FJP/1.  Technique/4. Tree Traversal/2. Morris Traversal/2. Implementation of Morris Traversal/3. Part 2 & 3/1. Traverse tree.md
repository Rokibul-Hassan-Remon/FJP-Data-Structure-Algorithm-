	function morrisTraversal(root):
	    current = root
	
	    while current is not NULL:
	
	        if current.left is NULL:
	            // ✅ Left নেই → Node visit → Right এ যাও
	            visit(current)
	            current = current.right
	
	        else:
	            // ✅ Left আছে → predecessor খুঁজো
	            predecessor = current.left
	
	            // ✅ predecessor = left subtree এর সবচেয়ে ডানপাশের node
	            while predecessor.right is NOT NULL 
	                  AND predecessor.right is NOT current:
	                predecessor = predecessor.right
	
	            if predecessor.right is NULL:
	                // ✅ প্রথমবার: Thread তৈরি করো
	                predecessor.right = current
	                current = current.left
	
	            else:
	                // ✅ দ্বিতীয়বার: Thread কেটে দাও
	                predecessor.right = NULL
	                visit(current)
	                current = current.right
